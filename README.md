import numpy as np
import matplotlib.pyplot as plt


def corr2cov(p: np.ndarray, s: np.ndarray) -> np.ndarray:
    """–ö–æ–≤–∞—Ä–∏–∞—Ü–∏–æ–Ω–Ω–∞—è –º–∞—Ç—Ä–∏—Ü–∞ –æ—Ç –∫–æ—Ä—Ä–µ–ª—è—Ü–∏–∏ –∏ —Å—Ç–∞–Ω–¥–∞—Ä—Ç–Ω—ã—Ö –æ—Ç–∫–ª–æ–Ω–µ–Ω–∏–π"""
    d = np.diag(s)
    return d @ p @ d

# 0.9 —ç—Ç–æ –∫–æ—Ä—Ä–µ–ª—è—Ü–∏—è –º–µ–∂–¥—É –ê –∏ B, –∞ –∫–æ—Ä—Ä–µ–ª—è—Ü–∏—è
# —Å–∞–º–æ–π –ø–µ—Ä–µ–º–µ–Ω–Ω–æ–π —Ä–∞–≤–Ω–∞ 1.0
corr = np.array([[1., 0.9],
                [0.9, 1.]])

stdev = np.array([3., 1.])
mean = np.array([5., -5.])
cov = corr2cov(corr, stdev)

data = np.random.multivariate_normal(mean=mean, cov=cov, size=5000)
